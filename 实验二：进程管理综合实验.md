### 任务一：初探进程裂变 —— `fork()` 的基础使用

**【知识讲解】**

在 Linux 中，创建新进程的唯一常用方法是调用 `fork()` 函数。它就像细胞分裂，调用一次，返回**两次**：

1. 给父进程返回新建子进程的 PID（一个大于 0 的整数）。
2. 给子进程返回 `0`。
3. 如果出错，返回 `-1`。

**【操作步骤】**

1. **建立实验目录：** 在你的家目录下创建一个新文件夹并进入。

   ```Bash
   $ mkdir os_lab2 && cd os_lab2
   ```

2. **编写代码：** 使用 `vim` 创建文件 `fork_test1.c`。

   ```Bash
   $ vim fork_test1.c
   ```

   *(提示：按 `i` 进入插入模式，输入以下代码。注意，指导书原代码中的 `=` 已修正为 `==` 以防止死循环。)*

   ```C
   #include <stdio.h>
   #include <unistd.h>
   #include <sys/wait.h> // 引入 wait() 函数的头文件
   
   int main() {
       int p1;
   
       // 关闭标准输出的缓冲区！
       // 这样每个 putchar 都会立刻直接送到屏幕上，
       // 既解决了 'x' 复制的问题，又让我们能真实看到操作系统的“实时竞速”。
       setbuf(stdout, NULL); 
   
       putchar('x');
       
       // 创建子进程
       while ((p1 = fork()) == -1); 
   
       if (p1 == 0) {
           // 子进程逻辑
           putchar('b');
       } else {
           // 父进程逻辑
           putchar('a');
       }
       
       // 父子进程都会执行这一句
       putchar('y');
   
       // 父进程等待，统一收尾
       if (p1 > 0) {
           wait(NULL);   // 父进程在这里停下，死等子进程执行完毕退出
           putchar('\n'); // 确认所有字符都打完了，父进程打上唯一的换行符
       }
   
       return 0;
   }
   ```

   *(按 `ESC`，输入 `:wq` 保存退出)*

3. **编译并运行：**

   ```Bash
   $ gcc fork_test1.c -o fork_test1
   $ ./fork_test1
   ```

**【观察与分析】**  请反复多次运行该程序（例如连续运行 10 次）。观察输出到底是 `xayby`、`xabyy` 还是其他组合？

### 任务二：多进程并发调度分析树

**【知识讲解】**

进程可以不断地产生子进程，形成一棵进程树。在这个任务中，我们将观察复杂的嵌套 `fork()`。

**【操作步骤】**

1. 创建并编辑 `fork_test2.c`：

   ```C
   #include <stdio.h>
   #include <unistd.h>
   #include <sys/wait.h>
   
   int main() {
       // 【关键修复 1】：必须赋初始值 0，防止未初始化的垃圾值干扰判断
       int p1 = 0, p2 = 0; 
       
       // 【关键修复 2】：关闭缓冲区，保证字符实时输出且不被复制
       setbuf(stdout, NULL);
   
       while ((p1 = fork()) == -1);
   
       if (p1 == 0) {
           // 这是 子进程 1 执行的逻辑
           putchar('b');
       } else {
           // 这是 父进程 执行的逻辑
           while ((p2 = fork()) == -1);
           
           if (p2 == 0) {
               // 这是 子进程 2 执行的逻辑
               // 注意：子进程 2 继承了 p1 的值（p1 > 0），但它自己 fork 得到的 p2 == 0
               putchar('c');
           } else {
               // 这是 真正的父进程 执行的逻辑
               // 此时：p1 > 0 且 p2 > 0
               putchar('a');
           }
       }
       
       // 【关键修复 3】：精确定位真正的父进程，并统一收尾
       // 只有真正的父进程，它的 p1 和 p2 才都会大于 0
       if (p1 > 0 && p2 > 0) {
           wait(NULL);    // 等待第一个死去的孩子
           wait(NULL);    // 等待第二个死去的孩子
           putchar('\n'); // 确认两个孩子都执行完了，打上唯一的换行符
       }
   
       return 0;
   }
   ```

2. 编译并反复运行 。

### 任务三：克隆体的独立性 —— 数据隔离验证

**【知识讲解】**

`fork()` 出来的子进程会完美复制父进程的内存空间（包括变量）。但是，它们拥有**各自独立的地址空间**。修改其中一个进程的变量，绝不会影响另一个进程（这被称为“写时复制” Copy-on-Write 机制）。

**【操作步骤】**

1. 创建并编辑 `fork_test3.c`：

   ```C
   #include <stdio.h>
   #include <unistd.h>
   #include <sys/wait.h> 
   
   int main() {
       int p1;
       int x = 1; // 定义一个初始变量
   
       setbuf(stdout, NULL);
   
       while ((p1 = fork()) == -1);
   
       if (p1 == 0) {
           // --- 子进程逻辑 ---
           putchar('b');
           x = 9;  // 子进程修改 x 的值
           printf(" (Child x=%d) ", x);
       } else {
           // --- 父进程逻辑 ---
           putchar('a');
           printf(" (Parent x=%d) ", x); 
       }
   
       // 只有父进程（p1 > 0）负责等待子进程，并打上最后的换行符
       if (p1 > 0) {
           wait(NULL);   // 父进程阻塞，等待子进程执行完毕
           printf("\n"); // 确认双方都输出完毕后，打上唯一的换行符
       }
   
       return 0;
   }
   ```

2. 编译并运行，重点观察子进程中对 `x` 重新赋值对父进程有什么影响 。

### 任务四：秩序的建立 —— 信号量与进程互斥

**【知识讲解】**

如果多个并发进程同时抢占终端输出字符，屏幕上就会出现混乱的交错打印。为了保证同一时刻只有一个进程能执行特定代码（进入临界区），我们需要引入信号量（Semaphore）机制。

- **P 操作（Wait）：** 尝试获取资源，如果信号量 <= 0，则进程阻塞休眠；如果 >0，则信号量减 1，进程继续执行。
- **V 操作（Signal）：** 释放资源，信号量加 1，并唤醒阻塞的进程。

**【操作步骤】** 本程序要求：父子进程循环进入临界区，必须是一个进程出来后，另一个才能进去 。  

1. 建立 `mutex_test.c`。
2. 将以下基于系统调用的信号量代码录入。**代码篇幅较长，请注意核对缩进和头文件。**

```C
#include <unistd.h>
#include <stdio.h>
#include <stdlib.h>
#include <sys/types.h>
#include <sys/ipc.h>
#include <sys/sem.h>

// 解决部分 Linux 系统下 semun 联合体未定义的问题
#ifndef SEMUN_H
#define _SEMUN_H
union semun {
    int val;
    struct semid_ds *buf;
    unsigned short int *array;
    struct seminfo *__buf;
};
#endif

/* 函数声明 */
static int set_mutex(void);      /* 信号量 mutex 赋初值函数 */
static void del_mutex(void);     /* 删除信号量 mutex 函数 */
static int p_mutex(void);        /* P(mutex)操作函数 */
static int v_mutex(void);        /* V(mutex)操作函数 */
static int sem_id;               /* 信号量全局标识 */

int main() {
    int chld, i, j;
    // 创建互斥信号量，键值为1234
    sem_id = semget((key_t)1234, 1, 0666 | IPC_CREAT); 
    set_mutex(); // 设置初值为1

    while ((chld = fork()) == -1);
    
    if (chld > 0) { 
        i = 1;
        while (i <= 3) {
            sleep(1); // 模拟进入临界区前的准备工作
            
            p_mutex(); // 【加锁】执行P操作，进入临界区
            printf("prnt in\n");
            sleep(1);  // 模拟在临界区内占用资源处理业务
            printf("prnt out\n");
            v_mutex(); // 【解锁】执行V操作，出临界区
            
            i++;
        }
        sleep(2); // 等待子进程执行完毕
        del_mutex(); // 父进程负责销毁信号量
        exit(0);
    } else { 
        j = 1;
        while (j <= 3) {
            sleep(1);
            
            p_mutex(); // 【加锁】执行P操作
            printf("chld in\n");
            sleep(1);  
            printf("chld out\n");
            v_mutex(); // 【解锁】执行V操作
            
            j++;
        }
        exit(0);
    }
}

/* 信号量赋初值函数：互斥信号初值为1 */
static int set_mutex(void) {
    union semun sem_union;
    sem_union.val = 1; 
    if (semctl(sem_id, 0, SETVAL, sem_union) == -1) return(1);
    return(0);
}

/* 删除信号量函数 */
static void del_mutex(void) {
    union semun sem_union;
    if (semctl(sem_id, 0, IPC_RMID, sem_union) == -1) {
        fprintf(stderr, "Failed to delete semaphore\n");
    }
}

/* 定义P操作函数：负数表示P操作，每次减1 */
static int p_mutex(void) {
    struct sembuf sem_b;
    sem_b.sem_num = 0;
    sem_b.sem_op = -1; 
    sem_b.sem_flg = SEM_UNDO;
    if (semop(sem_id, &sem_b, 1) == -1) {
        fprintf(stderr, "p_mutex failed\n");
        return(0);
    }
    return(1);
}

/* 定义V操作函数：正数表示V操作，每次加1 */
static int v_mutex(void) {
    struct sembuf sem_b;
    sem_b.sem_num = 0;
    sem_b.sem_op = 1;  
    sem_b.sem_flg = SEM_UNDO;
    if (semop(sem_id, &sem_b, 1) == -1) {
        fprintf(stderr, "v_mutex failed\n");
        return(0);
    }
    return(1);
}
```
